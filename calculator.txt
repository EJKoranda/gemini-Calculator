<script>
let display = document.getElementById('display');

function appendToDisplay(value) {
  display.value += value;
}

function clearDisplay() {
  display.value = '';
}

function calculate() {
  try {
    display.value = eval(display.value); // Use eval cautiously!
  } catch (error) {
    display.value = 'Error';
  }
}
</script>
